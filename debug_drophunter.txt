    <script>
        function showDemoModal() {
            const modal = new bootstrap.Modal(document.getElementById('demoModal'));
            modal.show();
        }
        
        document.addEventListener('DOMContentLoaded', function() {
            // Track demo page interactions
            console.log('Drop Hunter demo page loaded');
            
            // Enable search functionality
            const searchInput = document.getElementById('campaignSearch');
            if (searchInput) {
                searchInput.addEventListener('input', function() {
                    const searchText = this.value.toLowerCase();
                    const rows = document.querySelectorAll('tbody tr');
                    
                    rows.forEach(row => {
                        const gameNameEl = row.querySelector('h6');
                        const campaignNameEl = row.querySelector('p strong');
                        
                        if (!gameNameEl) return; // Skip if not a campaign row
                        
                        const gameName = gameNameEl.textContent.toLowerCase();
                        const campaignName = campaignNameEl ? campaignNameEl.parentElement.textContent.toLowerCase() : '';
                        
                        const matches = gameName.includes(searchText) || 
                                      campaignName.includes(searchText);
                        
                        row.style.display = matches ? '' : 'none';
                    });
                });
            }
        });
    </script>
